DROP TABLE IF EXISTS Sports;
CREATE TABLE Sports(
	SportId int identity(1,1),
	Name Varchar(50) NOT NULL,
	IsEnabled bit NOT NULL,
 	CONSTRAINT PK_Sports PRIMARY KEY 
	(
		SportId 
	) 
);
